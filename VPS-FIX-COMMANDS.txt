VPS MongoDB Trial Settings Fix - Copy These Commands

# Step 1: Add MongoDB force flag
echo "FORCE_MONGODB=true" >> .env

# Step 2: Update DATABASE_URL (replace with your MongoDB URL)
# Edit .env file and update DATABASE_URL line:
nano .env

# Replace DATABASE_URL line with your MongoDB connection:
DATABASE_URL=mongodb+srv://username:password@cluster.mongodb.net/jellyfin_signup

# Step 3: Restart application
docker-compose down
docker-compose up -d

# Step 4: Verify MongoDB detection
curl http://localhost:5000/health

# Should show "databaseType": "MongoDB"

# Step 5: Check storage logs
docker-compose logs | grep "Storage Configuration"

# Should show "Using storage: MongoDB"
